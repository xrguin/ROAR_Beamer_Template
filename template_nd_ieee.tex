% ---- Slide 3: equations, IEEE-style ----
\begin{frame}{IEEE-style equations}
  Equations are part of the sentence flow and end with the appropriate
  punctuation. Define the posterior over the stochastic latent as
  \begin{equation}\label{eq:posterior}
    s_t \sim q_\phi(s_t \mid h_t, o_t),
  \end{equation}
  and decompose the world-model objective into a reconstruction term and a
  KL term,
  \begin{align}
    \mathcal{L}_{\mathrm{rec}}
    &= -\E_{q_\phi}\!\left[\log p_\theta(o_t \mid h_t, s_t)\right],
    \label{eq:rec}
    \\
    \mathcal{L}_{\mathrm{KL}}
    &= \KL\!\big(q_\phi(s_t \mid h_t, o_t)\,\big\|\,p_\theta(s_t \mid h_t)\big).
    \label{eq:kl}
  \end{align}
  Combining~\eqref{eq:rec} and~\eqref{eq:kl} with~\eqref{eq:posterior} gives
  the ELBO used during training.

  \medskip
  Inline conventions: bold lowercase italic for vectors $\bm{x}\in\mathbb{R}^{n}$,
  bold capital italic for matrices $\bm{A}\in\mathbb{R}^{n\times n}$, upright
  roman subscripts for word labels ($\mathcal{L}_{\mathrm{rec}}$ not
  $\mathcal{L}_{rec}$), optimizers via $\argmin_{\theta} \mathcal{L}(\theta)$.
\end{frame}
